    HOSPITAL {
      int hospital_id PK
      string name
      string location
      string contact
    }

    PATIENT {
      int patient_id PK
      int hospital_id FK
      string full_name
      int age
      string gender
      string blood_group
      string contact_info
      decimal risk_score
      datetime created_at
    }

    MEDICAL_RECORD {
      int record_id PK
      int patient_id FK
      string diagnosis
      string severity_level
      string stage
      decimal hemoglobin_level
      datetime updated_at
    }

    DONOR {
      int donor_id PK
      int hospital_id FK
      string full_name
      int age
      string blood_group
      string donor_type
      string contact_info
      string availability_status
      datetime last_donation_date
    }

      int request_id PK
      int patient_id FK

      string request_type
      int urgency_level
      decimal priority_score
      string status
      datetime request_date
    }

    BLOOD_REQUEST_DETAILS {
      int request_id PK,FK
      string blood_group_required
      int units_required
      datetime required_by
    }

      int request_id PK,FK
      string organ_type_required

      string hla_notes
    }

    BLOOD_UNIT {
